# 4. supplyBase withdrawBase回归主线

## `supplyBase()`

```solidity
function supplyBase(address from, address dst, uint256 amount) internal {
        //用户把 amount 的 Token 转进来
        amount = doTransferIn(baseToken, from, amount);

        accrueInternal(); //更新协议利息,更新index

        //找到用户当前的 base 借贷净仓位 principal
        UserBasic memory dstUser = userBasic[dst];
        int104 dstPrincipal = dstUser.principal; //// int104，有正负
        int256 dstBalance = presentValue(dstPrincipal) + signed256(amount); //计算存款之后你真正的当前余额,就是把你当前的真实余额（正负都可以）加上本次存的钱，得到一个新的真实余额
        int104 dstPrincipalNew = principalValue(dstBalance); //把这个新的 balance 转回 principal（本金）,作用就是把真实余额除以最新 index → 得到新的本金

        //重要
        (uint104 repayAmount, uint104 supplyAmount) = repayAndSupplyAmount(dstPrincipal, dstPrincipalNew); //拆分amount的repay和supply部分

        //全局状态更新
        totalSupplyBase += supplyAmount;
        totalBorrowBase -= repayAmount;

        updateBasePrincipal(dst, dstUser, dstPrincipalNew);

        emit Supply(from, dst, amount);

        if (supplyAmount > 0) {
            emit Transfer(address(0), dst, presentValueSupply(baseSupplyIndex, supplyAmount));
        }
    }
```

1. 用户把 base token 转入协议
2. `accrueInternal()`：先把全局 index 推到当前时刻
3. 读取用户旧 `principal`
4. `presentValue(oldPrincipal)`：把旧 principal 还原成当前真实余额
5. `newBalance = oldPresentValue + amount`
6. `principalValue(newBalance)`：把新的真实余额压回 principal
7. 拆分这次变化中有多少是 `repayAmount`、多少是 `supplyAmount`
8. 更新全局 `totalBorrowBase / totalSupplyBase`
9. `updateBasePrincipal(...)`：写回用户 principal 并结算 tracking 奖励

## `withdrawBase()`

```solidity
 function withdrawBase(address src, address to, uint256 amount) internal {
        accrueInternal(); //更新全局利率

        UserBasic memory srcUser = userBasic[src];
        int104 srcPrincipal = srcUser.principal; //读取旧余额
        int256 srcBalance = presentValue(srcPrincipal) - signed256(amount); //新=旧-amount
        int104 srcPrincipalNew = principalValue(srcBalance); //把新的真实余额重新压回 principal,注意有正负

        (uint104 withdrawAmount, uint104 borrowAmount) = withdrawAndBorrowAmount(srcPrincipal, srcPrincipalNew);

        totalSupplyBase -= withdrawAmount;
        totalBorrowBase += borrowAmount;

        updateBasePrincipal(src, srcUser, srcPrincipalNew); //更新用户 principal

        if (srcBalance < 0) {
            //如果进入借款状态，要做风控检查
            if (uint256(-srcBalance) < baseBorrowMin) revert BorrowTooSmall(); //是否满足最小借款规模
            if (!isBorrowCollateralized(src)) revert NotCollateralized(); //是否足够抵押
        }

        doTransferOut(baseToken, to, amount); //把 base token 转出去

        emit Withdraw(src, to, amount);

        if (withdrawAmount > 0) {
            emit Transfer(src, address(0), presentValueSupply(baseSupplyIndex, withdrawAmount));
        }
    }
```

1. `accrueInternal()`：先把全局 index 推到当前时刻
2. 读取用户旧 `principal`
3. `presentValue(oldPrincipal)`：把旧 principal 还原成当前真实余额
4. `newBalance = oldPresentValue - amount`
5. `principalValue(newBalance)`：把新的真实余额压回 principal
6. 拆分这次变化中有多少是 `withdrawAmount`、多少是 `borrowAmount`
7. 更新全局 `totalSupplyBase / totalBorrowBase`
8. `updateBasePrincipal(...)`：写回用户 principal 并结算 tracking 奖励
9. 若操作后进入借款状态，则检查：
   * 借款是否达到最小规模
   * 抵押是否充足
10. `doTransferOut(...)`：真正把 base token 转给用户

## `repayAndSupplyAmount()`

用于 `supplyBase()`

```solidity
function repayAndSupplyAmount(int104 oldPrincipal, int104 newPrincipal) internal pure returns (uint104, uint104) {
        // If the new principal is less than the old principal, then no amount has been repaid or supplied
        if (newPrincipal < oldPrincipal) return (0, 0);//说明这次 principal 变化方向不对，不属于“还债/新增存款”这个方向

        if (newPrincipal <= 0) {//新 principal 还在负数区或者刚好到 0
            return (uint104(newPrincipal - oldPrincipal), 0);
        } else if (oldPrincipal >= 0) {//旧 principal 本来就在非负区
            return (0, uint104(newPrincipal - oldPrincipal));
        } else {//既还清债务，又开始存款
            return (uint104(-oldPrincipal), uint104(newPrincipal));
        }
    }
```

## `withdrawAndBorrowAmount()`

用于 `withdrawBase()`

```solidity
 function withdrawAndBorrowAmount(int104 oldPrincipal, int104 newPrincipal)
        internal
        pure
        returns (uint104, uint104)
    {
        // If the new principal is greater than the old principal, then no amount has been withdrawn or borrowed
        if (newPrincipal > oldPrincipal) return (0, 0); //说明这不是 withdraw/borrow 方向

        if (newPrincipal >= 0) {
            //仍然在非负区,只是从存款里取了一部分,还没进入借款
            return (uint104(oldPrincipal - newPrincipal), 0);
        } else if (oldPrincipal <= 0) {
            //旧 principal 本来就在负数区或 0,即在原有借款基础上借得更多
            return (0, uint104(oldPrincipal - newPrincipal));
        } else {
            //之前的存款全提完而且还新增了借款
            return (uint104(oldPrincipal), uint104(-newPrincipal));
        }
    }
```

### 为什么是用 principal 拆，而不是用真实余额拆?

全局存的就是：

* `totalSupplyBase`
* `totalBorrowBase`

也就是 **principal 维度** 的总量。

所以拆分也要在 principal 维度做。

然后等你需要真实总供给/总借款时，再配合 index 去投影：

* `presentValueSupply(baseSupplyIndex,totalSupplyBase)`
* `presentValueBorrow(baseBorrowIndex,totalBorrowBase)`

这和前面整套设计是完全一致的。
