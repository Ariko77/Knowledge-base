# 下载一些包需要的指令

下载一些包需要的指令  
forge init   
forge install OpenZeppelin/openzeppelin-contracts  --no-commit（ERC一类的东西可以直接用）  
forge install huff-language/huffmate （huff math)  
remappings = ["@openzeppelin/contracts=lib/openzeppelin-contracts/contracts"]

--with-gas-price 0 --priority-gas-price 0
