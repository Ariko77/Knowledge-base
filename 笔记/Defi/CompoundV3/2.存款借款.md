最终都会到`supplyInternal()`：

```solidity
function supplyInternal(address operator, address from, address dst, address asset, uint256 amount)
        internal
        nonReentrant
    {
        if (isSupplyPaused()) revert Paused();
        if (!hasPermission(from, operator)) revert Unauthorized();

        if (asset == baseToken) {
            //如果 asset 是 base asset → 增加你的 borrow balance 的健康度
            if (amount == type(uint256).max) {
                amount = borrowBalanceOf(dst);
            }
            return supplyBase(from, dst, amount);
        } else {
            //如果 asset 是 collateral → 增加抵押物价值
            return supplyCollateral(from, dst, asset, safe128(amount));
        }
    }
 
```

* \*\*base asset（比如 USDC） → \*\*<code>**supplyBase()**</code>
* \*\*抵押物（例如 WETH） → \*\*<code>**supplyCollateral()**</code>

## \*\*抵押base asset => \*\*`supplyBase()`

```solidity
   /**
     * @dev Supply an amount of base asset from `from` to dst
     */
    function supplyBase(address from, address dst, uint256 amount) internal {
      //用户把 amount 的 Token 转进来  
      amount = doTransferIn(baseToken, from, amount);

        accrueInternal();//

        UserBasic memory dstUser = userBasic[dst];
        int104 dstPrincipal = dstUser.principal;
        int256 dstBalance = presentValue(dstPrincipal) + signed256(amount);
        int104 dstPrincipalNew = principalValue(dstBalance);

        (uint104 repayAmount, uint104 supplyAmount) = repayAndSupplyAmount(dstPrincipal, dstPrincipalNew);

        totalSupplyBase += supplyAmount;
        totalBorrowBase -= repayAmount;

        updateBasePrincipal(dst, dstUser, dstPrincipalNew);

        emit Supply(from, dst, amount);

        if (supplyAmount > 0) {
            emit Transfer(address(0), dst, presentValueSupply(baseSupplyIndex, supplyAmount));
        }
    }

   
```

Compound V3 把用户的 base position 压缩成一个 “principal”

* `principal > 0` = 用户是 **存款人**（lender）
* `principal < 0` = 用户是 **借款人**（borrower）

本金并不是余额，而是一个经过 index 缩放的数字。

真正的当前余额要用：

```solidity
presentValue(principal)
```

如果 principal > 0（存款）：

```solidity
presentValue = principal * supplyIndex
```

如果 principal < 0（借款）：

```solidity
presentValue = principal * borrowIndex
```

（借款指数永远比存款指数涨得快）

***

这里计算存款之后的真实余额：

```solidity
int256 dstBalance = presentValue(dstPrincipal) + amount;
```

* 你现在借了 100（balance = -100）
* 你存 30
* 新余额 = -100 + 30 = -70（依然欠钱）

### `principalValue(principal)`

```solidity
function principalValue(int256 presentValue_) internal view returns (int104) {
        if (presentValue_ >= 0) {
            return signed104(principalValueSupply(baseSupplyIndex, uint256(presentValue_)));
        } else {
            return -signed104(principalValueBorrow(baseBorrowIndex, uint256(-presentValue_)));
        }
    }
```

* 如果 principal ≥ 0（存款人） → 用 supplyIndex

realBalance = principal \* supplyIndex

* 如果 principal < 0（借款人） → 用 borrowIndex

realBalance = principal \* borrowIndex

### principalValue(realBalance) 反推本金

principal = realBalance / index

注意 realBalance 也可能是负的

**总结**

* supplyIndex 随时间变大（存款赚利息）
* borrowIndex 随时间变更快的变大（借款付利息）

而本金 principal 本身不会动。

这意味着：

用户的实际余额只由指数变化自动更新，不需要触碰 userBasic 数据。

### repayAndSupplyAmount 重要逻辑

```solidity
(repayAmount, supplyAmount) = repayAndSupplyAmount(dstPrincipal, dstPrincipalNew);
```

本次supply以后结果如何

#### 你原本是负的 borrower

例：\
你原来 principal 表示你借了 100\
现在你 supply 了 150

你的新 balance = -100 + 150 = 50

所以：

* 其中 100 是 **偿还借款**
* 之后剩 50 是 **新的存款**

因此：

```solidity
repayAmount  = 100
supplyAmount = 50
```

这就是：

```solidity
totalSupplyBase += supplyAmount;   // 新增存款金额
totalBorrowBase -= repayAmount;    // 减少借款金额
```

#### 你原来是存款人

例：\
你本来有 200 存款\
你再存 50

那：

```solidity
repayAmount  = 0
supplyAmount = 50
```

借款总额不变，只增加 supply

#### 总结

设原余额为 B\_old\
设存入金额为 A\
新余额：

```solidity
B_new = B_old + A
```

如果 B\_old < 0（借款人）：

```solidity
repayAmount  = min(A, -B_old)
supplyAmount = A - repayAmount
```

如果 B\_old ≥ 0（存款人）：

```solidity
repayAmount  = 0
supplyAmount = A
```

***

```solidity
otalSupplyBase += supplyAmount
totalBorrowBase -= repayAmount
```

<font style="background-color:#E6DCF9;">repayAmount 让系统借款减少</font>

<font style="background-color:#E6DCF9;">supplyAmount 增加系统存款</font>       => 这个理解一下,就很好明白他怎么变的了

## 抵押物 => `supplyCollateral()`

* **base asset（USDC）**
  * 有利息
  * 有 借 / 还 / 存 三态
  * 用 **principal + index** 压缩状态
* **collateral（WETH 等）**
  * 没利息
  * 不借不还
  * 只参与 **风险约束（能不能借）**

**所以 collateral 的数学重点不在“余额变化”，而在：**

它如何影响你“最多能借多少 base”

```solidity
/**
     * @dev Supply an amount of collateral asset from `from` to dst
     */
    //抵押物路径
    function supplyCollateral(address from, address dst, address asset, uint128 amount) internal {
        //用户转钱
        //safe128 用于把 uint256 安全转换成 uint128，保证不会溢出（抵押物数量不会超过 2^128）
        amount = safe128(doTransferIn(asset, from, amount));
        //返回值是实际转入的数量（防止 token 有 transfer fee）

        AssetInfo memory assetInfo = getAssetInfoByAddress(asset);//这个 token 的信息，比如价格、抵押率、供给上限。
        TotalsCollateral memory totals = totalsCollateral[asset];//totalsCollateral:这个 token 的总抵押物信息
        totals.totalSupplyAsset += amount;//系统总抵押量增加
        if (totals.totalSupplyAsset > assetInfo.supplyCap) revert SupplyCapExceeded();//上限

        uint128 dstCollateral = userCollateral[dst][asset].balance;//用户当前的这个抵押物数量
        uint128 dstCollateralNew = dstCollateral + amount;//存入之后的新数量

        totalsCollateral[asset] = totals;//更新协议总量
        userCollateral[dst][asset].balance = dstCollateralNew;//更新用户自己的抵押量

        //更新用户的活跃资产列表
        //如果用户第一次存某种抵押物，它会被加入 assetsIn[dst]，标记为活跃抵押物。
        //这样系统知道这个用户可以用它来借钱
        updateAssetsIn(dst, assetInfo, dstCollateral, dstCollateralNew);

        emit SupplyCollateral(from, dst, asset, amount);
    }
```

| 数学量 | 含义 |
| --- | --- |
| `userCollateral[dst][asset].balance` | 你存了多少 token |
| `price` | 这个 token 的价格 |
| `borrowCollateralFactor` | 抵押折扣率（LTV） |

真正参与计算的是：

```solidity
有效抵押价值 = balance × price × collateralFactor
```

collateral 不会参与指数，不会自动变化，全是线性计算

### `updateAssetsIn()`

```solidity
function updateAssetsIn(
        address account,
        AssetInfo memory assetInfo,
        uint128 initialUserBalance,
        uint128 finalUserBalance
    ) internal {
        if (initialUserBalance == 0 && finalUserBalance != 0) {
            // set bit for asset
            userBasic[account].assetsIn |= (uint16(1) << assetInfo.offset);
        } else if (initialUserBalance != 0 && finalUserBalance == 0) {
            // clear bit for asset
            userBasic[account].assetsIn &= ~(uint16(1) << assetInfo.offset);
        }
    }
```

检查第 `i` 个资产是不是在这个账户的活跃抵押物集合里。

### `isBorrowCollateralized()`

```solidity
function isBorrowCollateralized(address account) public view override returns (bool) {
        int104 principal = userBasic[account].principal;

        if (principal >= 0) {
            // 当前没有base借款
            return true; // 可以借的/collateralized 的
            // 连负债都没有，风控上不可能不够抵押
        }

        //以下else
        uint16 assetsIn = userBasic[account].assetsIn; //把你当前有哪些抵押物在参与风控拿出来
        int256 liquidity = signedMulPrice(presentValue(principal), getPrice(baseTokenPriceFeed), uint64(baseScale)); //你的 base 头寸的价值
        //这里是先把你的抵押值记成负的

        //遍历抵押物，把“可借价值”一项项加回来
        for (uint8 i = 0; i < numAssets;) {
            if (
                isInAsset(assetsIn, i) //这个资产如果余额为 0，就跳过
                //如果余额非 0，就拿出来参与风控计算
            ) {
                if (liquidity >= 0) {
                    return true;//如果前面已经靠之前若干个抵押物把负债“补正了”，那就不用继续遍历了，直接返回 true
                }

                AssetInfo memory asset = getAssetInfo(i);
                //把某个 collateral 的余额换成价值,不是具体多少钱,是统一的价值
                uint256 newAmount =
                    mulPrice(userCollateral[account][asset.asset].balance, getPrice(asset.priceFeed), asset.scale);
                liquidity += signed256(mulFactor(newAmount, asset.borrowCollateralFactor));
            }
            unchecked {
                i++;
            }
        }

        return liquidity >= 0;
    }
```

先看principal,如果大于0,说明你当前没有base借款,直接返回true

```solidity
 if (principal >= 0) {
            // 当前没有base借款
            return true; // 可以借的/collateralized 的
            // 连负债都没有，风控上不可能不够抵押
        }
```

如果principal小于0,说明你当前有base借款,来看源码逻辑:

```solidity
//以下else
uint16 assetsIn = userBasic[account].assetsIn;//把你当前有哪些抵押物在参与风控拿出来
int256 liquidity = signedMulPrice(presentValue(principal), getPrice(baseTokenPriceFeed), uint64(baseScale));
```

拆开来看

* `presentValue(principal)`： 把压缩后的 `principal` 还原成当前真实的 base balance。\
  如果你是借款人，这个值是负的。
* `getPrice(baseTokenPriceFeed)`

拿 base token 的价格。

* `signedMulPrice(...)`

把“带符号的 token 数量”乘价格，变成带符号的价值。\
源码里它本质就是：

```solidity
function signedMulPrice(int256 n, uint256 price, uint64 fromScale) internal pure returns (int256) {
        return n * signed256(price) / int256(uint256(fromScale));
    }
```

**数量 × 价格 ÷ 精度**。

这个liquidity最终算出来是\*\*你的 base 头寸的价值，\*\*如果你借了 base，那么它就是一个负值。

上边先记成负的，接下来就计算你的抵押物价值，再一项项加回来

```solidity
 //遍历抵押物，把“可借价值”一项项加回来
        for (uint8 i = 0; i < numAssets;) {
            if (
                isInAsset(assetsIn, i) //这个资产如果余额为 0，就跳过
                //如果余额非 0，就拿出来参与风控计算
            ) {
                if (liquidity >= 0) {
                    return true;//如果前面已经靠之前若干个抵押物把负债“补正了”，那就不用继续遍历了，直接返回 true
                }

                AssetInfo memory asset = getAssetInfo(i);
                //把某个 collateral 的余额换成价值,不是具体多少钱,是统一的价值
                uint256 newAmount =
                    mulPrice(userCollateral[account][asset.asset].balance, getPrice(asset.priceFeed), asset.scale);
                liquidity += signed256(mulFactor(newAmount, asset.borrowCollateralFactor));
            }
            unchecked {
                i++;
            }
        }
```

抵押物数量 × 抵押物价格 ÷ 抵押物精度

```solidity
unction mulPrice(uint256 n, uint256 price, uint64 fromScale) internal pure returns (uint256) {
        return n * price / fromScale;
    }
```

**不是整个价值都能拿来借，要乘**<code>**borrowCollateralFactor**</code>

```solidity
liquidity += signed256(mulFactor(newAmount, asset.borrowCollateralFactor));
```

```solidity
function mulFactor(uint256 n, uint256 factor) internal pure returns (uint256) {
        return n * factor / FACTOR_SCALE;
    }
```

价值 × collateral factor

最后判断**liquidity**，大于0返回true，小于0返回false

```solidity
return liquidity >= 0;
```

#### borrowCollateralFactor 的含义

抵押物不是按 100% 价值计入借款能力。\
真正能支撑借款的是：

```solidity
有效抵押价值 = 抵押物价值 × borrowCollateralFactor
```

例如：

* WETH 价值 1000
* `borrowCollateralFactor = 0.8`

则最多只按 800 的借款能力计算。

#### `isBorrowCollateralized()`和`isLiquidatable()`的区别

* `isBorrowCollateralized(account)`\
  检查的是：**有没有足够流动性去借**, 是“借钱时给你的额度标准”
* `isLiquidatable(account)`\
  检查的是：**是不是已经差到可以被清算**, 是“清算时的底线标准”

借款检查用的是

```solidity
asset.borrowCollateralFactor
```

清算检查用的是

```solidity
asset.liquidateCollateralFactor
```

而且 `isLiquidatable` 最后返回的是：

```solidity
return liquidity < 0;
```

也就是它在看“清算口径下的流动性是不是已经负了”。

最终都会到`withdrawInternal()`:

```solidity
function withdrawInternal(address operator, address src, address to, address asset, uint256 amount)
internal
nonReentrant
{
  if (isWithdrawPaused()) revert Paused(); //检查 pause,就是协议的小总开关
  if (!hasPermission(src, operator)) revert Unauthorized(); //检查 operator 权限

  //判断你取的是不是 baseToken
  if (asset == baseToken) {
    if (amount == type(uint256).max) {
      amount = balanceOf(src);
    }
    return withdrawBase(src, to, amount); //是 base 就走 withdrawBase
  } else {
    return withdrawCollateral(src, to, asset, safe128(amount)); //不是 base 就走 withdrawCollateral
```

## 取出 base asset => `withdrawBase()`

```solidity
function withdrawBase(address src, address to, uint256 amount) internal {
        accrueInternal(); //更新全局利率

        UserBasic memory srcUser = userBasic[src];
        int104 srcPrincipal = srcUser.principal; //读取旧余额
        int256 srcBalance = presentValue(srcPrincipal) - signed256(amount); //新=旧-amount
        int104 srcPrincipalNew = principalValue(srcBalance); //把新的真实余额重新压回 principal,注意有正负

        (uint104 withdrawAmount, uint104 borrowAmount) = withdrawAndBorrowAmount(srcPrincipal, srcPrincipalNew);

        totalSupplyBase -= withdrawAmount;
        totalBorrowBase += borrowAmount;

        updateBasePrincipal(src, srcUser, srcPrincipalNew); //更新用户 principal

        if (srcBalance < 0) {
            //如果进入借款状态，要做风控检查
            if (uint256(-srcBalance) < baseBorrowMin) revert BorrowTooSmall(); //是否满足最小借款规模
            if (!isBorrowCollateralized(src)) revert NotCollateralized(); //是否足够抵押
        }

        doTransferOut(baseToken, to, amount); //把 base token 转出去

        emit Withdraw(src, to, amount);

        if (withdrawAmount > 0) {
            emit Transfer(src, address(0), presentValueSupply(baseSupplyIndex, withdrawAmount));
        }
    }
```

数学本质:

```solidity
新真实余额 = 旧真实余额 - amount
```

如果这个新余额：

* 仍然 ≥ 0\
  → 你只是把自己之前存的 base 取出来
* 变成 < 0\
  → 你已经进入借款状态了

### `withdrawBase` 数学拆分

设：

* `B_old` = withdraw 前真实 base balance
* `A` = withdraw amount
* `B_new = B_old - A`

则：

#### 从原 base 存款中提走的部分

```solidity
withdrawAmount = min(A, max(B_old, 0))
```

#### 超出原存款、形成新增借款的部分

```solidity
borrowAmount = A - withdrawAmount
```

含义：

* 若 `B_old > 0`，先从正的 base 存款中扣
* 若不够，再把剩余部分记为借款
* 若 `B_old <= 0`，则整笔都属于新增借款。官方文档说明 base asset can be borrowed using withdraw。

#### 对全局量的影响

```solidity
totalSupplyBase -= withdrawAmount
totalBorrowBase += borrowAmount
```

与 `supplyBase` 完全镜像：

```solidity
supplyBase:
totalBorrowBase -= repayAmount
totalSupplyBase += supplyAmount
```

## 取出物 => `withdrawCollateral()`

减少某种 collateral 的余额，并把 token 转给用户。\
但提取后账户仍必须满足借款抵押要求，否则 revert。

```solidity
 function withdrawCollateral(address src, address to, address asset, uint128 amount) internal {
        uint128 srcCollateral = userCollateral[src][asset].balance;//拿到用户当前余额
        uint128 srcCollateralNew = srcCollateral - amount;//算出新余额

        totalsCollateral[asset].totalSupplyAsset -= amount;
        userCollateral[src][asset].balance = srcCollateralNew;//更新协议总量和用户余额

        AssetInfo memory assetInfo = getAssetInfoByAddress(asset);
        updateAssetsIn(src, assetInfo, srcCollateral, srcCollateralNew);//更新某种资产是非0还是0了

        // Note: no accrue interest, BorrowCF < LiquidationCF covers small changes
        if (!isBorrowCollateralized(src)) revert NotCollateralized();//重新看账户还够不够 collateralized

        doTransferOut(asset, to, amount);

        emit WithdrawCollateral(src, to, asset, amount);
    }
```
