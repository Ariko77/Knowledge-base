# 1. 存款 赎回 借款 还款 清算

# 一、存款 mint

## 用户外部调用`mint()`

```solidity
function mint(uint256 mintAmount) external override returns (uint256) {
        mintInternal(mintAmount);
        return NO_ERROR;
    }
```

## Ctoken的`mintinternal()`

```solidity
//当用户外部调用mint()时,就会到这里,至于为什么是这么设计,当然是因为省gas
    function mintInternal(uint256 mintAmount) internal nonReentrant {
        accrueInterest(); //让当前区块的利息结算到 totalBorrows、totalReserves、borrowIndex 等状态变量中。
        // mintFresh emits the actual Mint event if successful and logs on errors, so we don't need to
        mintFresh(msg.sender, mintAmount);
    }
```

## Ctoken的`mintfresh()`

```solidity
 function mintFresh(address minter, uint256 mintAmount) internal {
        /* Fail if mint not allowed */
        uint256 allowed = comptroller.mintAllowed(address(this), minter, mintAmount);
        if (allowed != 0) {
            //comptroller验证
            revert MintComptrollerRejection(allowed);
        }

        /* Verify market's block number equals current block number */
        //确保刚才 accrueInterest() 已经更新到了当前区块
        if (accrualBlockNumber != getBlockNumber()) {
            revert MintFreshnessCheck();
        }

        //获取汇率
        //这里的exchageRate就是每 1 个 cToken 对应多少标的资产
        //随着时间推移，exchangeRate 会变大,因为利息在积累
        Exp memory exchangeRate = Exp({mantissa: exchangeRateStoredInternal()});

        /////////////////////////
        // EFFECTS & INTERACTIONS
        // (No safe failures beyond this point)

        /*
         *  We call `doTransferIn` for the minter and the mintAmount.
         *  Note: The cToken must handle variations between ERC-20 and ETH underlying.
         *  `doTransferIn` reverts if anything goes wrong, since we can't be sure if
         *  side-effects occurred. The function returns the amount actually transferred,
         *  in case of a fee. On success, the cToken holds an additional `actualMintAmount`
         *  of cash.
         */
        uint256 actualMintAmount = doTransferIn(minter, mintAmount); //这行是真正把用户的钱转进来的地方
        //actualMintAmount 考虑了手续费（有些代币会扣 fee，比如 USDT、USDC），所以可能比你输入的少一点

        /*
         * We get the current exchange rate and calculate the number of cTokens to be minted:
         *  mintTokens = actualMintAmount / exchangeRate
         */

        //计算铸造的 cToken 数量
        //cToken 数量 = 存入资产 / 汇率
        //变化的是汇率,比如赎回时候比存入时的汇率变大了,ctoken数量不变,那资产就变多了
        uint256 mintTokens = div_(actualMintAmount, exchangeRate);

        /*
         * We calculate the new total supply of cTokens and minter token balance, checking for overflow:
         *  totalSupplyNew = totalSupply + mintTokens
         *  accountTokensNew = accountTokens[minter] + mintTokens
         * And write them into storage
         */
        //更新账本
        totalSupply = totalSupply + mintTokens; //所有人 cToken 的总量
        accountTokens[minter] = accountTokens[minter] + mintTokens; //当前用户的 cToken 余额

        /* We emit a Mint event, and a Transfer event */
        emit Mint(minter, actualMintAmount, mintTokens); //记录用户 mint 了多少
        emit Transfer(address(this), minter, mintTokens); //符合 ERC20 标准（从合约转给用户）

        /* We call the defense hook */
        // unused function
        // comptroller.mintVerify(address(this), minter, actualMintAmount, mintTokens);
    }

```

## Comptroller的`mintallowed()`

```solidity
function mintAllowed(address cToken, address minter, uint256 mintAmount) external override returns (uint256) {
        // Pausing is a very serious situation - we revert to sound the alarms
        require(!mintGuardianPaused[cToken], "mint is paused");
        //Guardian 模式。
        //在 Compound 升级中，增加了一个“紧急开关”（guardian），当出现漏洞或风险时，可以暂停某个市场的 mint 行为

        // Shh - currently unused
        minter;
        mintAmount; //用来“避免未使用变量警告”，这里暂时没逻辑意义

        if (!markets[cToken].isListed) {
            //检查市场是否被列入
            return uint256(Error.MARKET_NOT_LISTED);
        }

        // Keep the flywheel moving
        updateCompSupplyIndex(cToken); //更新该市场的 COMP 奖励累计指数；
        distributeSupplierComp(cToken, minter); //给当前存钱的用户 minter 发放他应得的 COMP

        return uint256(Error.NO_ERROR);
        //只要没被暂停、市场存在、分发成功，就返回 0（NO_ERROR）。
        //否则，cToken 合约那边会收到错误码并中止执行
    }
```

## 流程

```solidity
用户 -> cToken.mint() 
       -> mintInternal()
          -> accrueInterest() 更新利息
          -> mintFresh()
               -> comptroller.mintAllowed() 权限检查 & COMP奖励
               -> exchangeRateStoredInternal() 获取汇率
               -> doTransferIn() 转入资产
               -> 计算 mintTokens
               -> 更新账户 & totalSupply
               -> emit Mint 事件
```

# 二、赎回 redeem

## 用户外部调用`redeem()`

```solidity
function redeem(uint redeemTokens) external returns (uint) {
        redeemInternal(redeemTokens);
        return NO_ERROR;
    }
```

## Ctoken的`redeemInternal()`或`redeemUnderlyingInternal()`

```solidity
//用户外部调用redeem就会到这里,用户输入想赎回多少 cToken
    //用户得到系统按汇率算出的 underlying
    function redeemInternal(uint256 redeemTokens) internal nonReentrant {
        accrueInterest(); //更新汇率
        // redeemFresh emits redeem-specific logs on errors, so we don't need to
        redeemFresh(payable(msg.sender), redeemTokens, 0);
    }
```

```solidity
//或者到这里,用户输入想赎回多少 Underlying Token
    //用户得到系统按汇率倒算出的 cToken 数
    function redeemUnderlyingInternal(uint256 redeemAmount) internal nonReentrant {
        accrueInterest(); //更新汇率
        // redeemFresh emits redeem-specific logs on errors, so we don't need to
        redeemFresh(payable(msg.sender), 0, redeemAmount);
    }
```

两者区别在于：

| 调用函数 | 用户输入 | 用户得到 | 用途 |
| --- | --- | --- | --- |
| redeemInternal | 想赎回多少 **cToken** | 系统按汇率算出 underlying | 常见方式 |
| redeemUnderlyingInternal | 想赎回多少 **Underlying Token** | 系统按汇率倒算出 cToken 数 | 精准提取金额 |

## Ctoken的`redeemFresh()`

```solidity
function redeemFresh(address payable redeemer, uint256 redeemTokensIn, uint256 redeemAmountIn) internal {
        require(redeemTokensIn == 0 || redeemAmountIn == 0, "one of redeemTokensIn or redeemAmountIn must be zero"); ////参数检查,要求只填一个参数,否则revert

        /* exchangeRate = invoke Exchange Rate Stored() */
        Exp memory exchangeRate = Exp({mantissa: exchangeRateStoredInternal()}); //获取汇率

        uint256 redeemTokens;
        uint256 redeemAmount;
        /* If redeemTokensIn > 0: */
        if (redeemTokensIn > 0) {
            //如果用户指定了要赎回多少 cToken
            /*
             * We calculate the exchange rate and the amount of underlying to be redeemed:
             *  redeemTokens = redeemTokensIn
             *  redeemAmount = redeemTokensIn x exchangeRateCurrent
             */
            redeemTokens = redeemTokensIn;
            redeemAmount = mul_ScalarTruncate(exchangeRate, redeemTokensIn);
            //就是「我要取回我这些 cToken 对应的本金 + 利息」
        } else {
            //如果用户指定了要取出多少 underlying
            /*
             * We get the current exchange rate and calculate the amount to be redeemed:
             *  redeemTokens = redeemAmountIn / exchangeRate
             *  redeemAmount = redeemAmountIn
             */
            redeemTokens = div_(redeemAmountIn, exchangeRate);
            redeemAmount = redeemAmountIn;
            //就是「我要提取 100 DAI，那我得销毁多少 cToken 呢？
        }

        /* Fail if redeem not allowed */
        uint256 allowed = comptroller.redeemAllowed(address(this), redeemer, redeemTokens);
        if (allowed != 0) {
            //确认这个市场允许赎回
            //检查这个人是否还有借款、会不会触发抵押不足
            //触发 COMP 奖励发放逻辑
            revert RedeemComptrollerRejection(allowed);
        }

        /* Verify market's block number equals current block number */
        if (accrualBlockNumber != getBlockNumber()) {
            //确保刚才已经结算到了当前区块
            revert RedeemFreshnessCheck();
        }

        /* Fail gracefully if protocol has insufficient cash */
        if (getCashPrior() < redeemAmount) {
            //确保合约内的现金足够兑付用户要赎回的 underlying
            revert RedeemTransferOutNotPossible();
        }

        /////////////////////////
        // EFFECTS & INTERACTIONS
        // (No safe failures beyond this point)

        /*
         * We write the previously calculated values into storage.
         *  Note: Avoid token reentrancy attacks by writing reduced supply before external transfer.
         */
        //重入防护,先改状态,再转钱
        totalSupply = totalSupply - redeemTokens;
        accountTokens[redeemer] = accountTokens[redeemer] - redeemTokens;

        /*
         * We invoke doTransferOut for the redeemer and the redeemAmount.
         *  Note: The cToken must handle variations between ERC-20 and ETH underlying.
         *  On success, the cToken has redeemAmount less of cash.
         *  doTransferOut reverts if anything goes wrong, since we can't be sure if side effects occurred.
         */
        //如果是 cDAI：调用 IERC20(DAI).transfer(redeemer, redeemAmount)
        //如果是 cETH：直接 redeemer.call{value: redeemAmount}("")
        doTransferOut(redeemer, redeemAmount);

        /* We emit a Transfer event, and a Redeem event */
        emit Transfer(redeemer, address(this), redeemTokens); //符合 ERC20 标准
        emit Redeem(redeemer, redeemAmount, redeemTokens); //自定义事件，记录操作细节

        /* We call the defense hook */
        //留空的hook（可以在升级时扩展安全逻辑）
        comptroller.redeemVerify(address(this), redeemer, redeemAmount, redeemTokens);
    }

```

## Comptroller的`redeemallowed()`

```solidity
function redeemAllowed(address cToken, address redeemer, uint256 redeemTokens)
        external
        override
        returns (uint256)
    {
        uint256 allowed = redeemAllowedInternal(cToken, redeemer, redeemTokens);
        if (allowed != uint256(Error.NO_ERROR)) {
            return allowed;
        }

        // Keep the flywheel moving
        updateCompSupplyIndex(cToken);
        distributeSupplierComp(cToken, redeemer);

        return uint256(Error.NO_ERROR);
```

```solidity
function redeemAllowedInternal(address cToken, address redeemer, uint256 redeemTokens)
        internal
        view
        returns (uint256)
    {
        if (!markets[cToken].isListed) {
            //检查市场是否存在
            return uint256(Error.MARKET_NOT_LISTED); //只有被 Comptroller 正式登记的市场（cToken）才能赎回
        }

        /* If the redeemer is not 'in' the market, then we can bypass the liquidity check */
        if (!markets[cToken].accountMembership[redeemer]) {
            //检查是否是市场成员
            //如果用户没有在该市场中有借款头寸（即不属于市场成员），那就直接放行赎回，不用查流动性。
            //如果只是存了钱,从来没借过,就直接放行
            return uint256(Error.NO_ERROR);
        }

        /* Otherwise, perform a hypothetical liquidity check to guard against shortfall */
        //它会计算「假如现在赎回 redeemTokens 数量的 cToken，会不会出现 shortfall（流动性不足）
        (Error err,, uint256 shortfall) =
            getHypotheticalAccountLiquidityInternal(redeemer, CToken(cToken), redeemTokens, 0);
        if (err != Error.NO_ERROR) {
            return uint256(err);
        }
        if (shortfall > 0) {
            //表示会影响流动性,不允许赎回
            return uint256(Error.INSUFFICIENT_LIQUIDITY);
        }

        return uint256(Error.NO_ERROR);
    }
```

## 流程

```solidity
用户 → cToken.redeem() 
      → redeemInternal()/redeemUnderlyingInternal()
          → accrueInterest()              // 更新利息
          → redeemFresh()
               → comptroller.redeemAllowed()   // 检查安全 & 更新奖励
               → exchangeRateStoredInternal()  // 计算汇率
               → 确定赎回数量 redeemTokens/redeemAmount
               → 检查现金是否足够 getCashPrior()
               → 更新账本 totalSupply/accountTokens
               → doTransferOut()               // 转出底层资产
               → emit Redeem/Transfer 事件
       └─ Comptroller.redeemVerify()           ✅ 验证传参合理性

```

# 存款和赎回对比

| 步骤 | mint | redeem |
| --- | --- | --- |
| 调用入口 | `mintInternal` | `redeemInternal` |
| 计算方向 | underlying → cToken | cToken → underlying |
| Comptroller 调用 | `mintAllowed()` | `redeemAllowed()` |
| 资金流向 | 用户 → cToken 合约 | cToken 合约 → 用户 |
| 更新字段 | `+ totalSupply` | `- totalSupply` |
| 更新账户 | `+ accountTokens` | `- accountTokens` |
| 事件 | Mint, Transfer | Redeem, Transfer |

## CToken.`borrowFresh()`

```solidity
//真正的借款执行函数
    function borrowFresh(address payable borrower, uint256 borrowAmount) internal {
        /* Fail if borrow not allowed */
        uint256 allowed = comptroller.borrowAllowed(address(this), borrower, borrowAmount);
        if (allowed != 0) {
            //市场是否被暂停
            //借款人是否在该市场；
            // 价格是否存在；
            // 借款上限（borrowCap）是否超标；
            // 最重要的：是否有足够抵押品
            revert BorrowComptrollerRejection(allowed);
        }

        /* Verify market's block number equals current block number */
        if (accrualBlockNumber != getBlockNumber()) {
            //确保 accrueInterest 已更新到最新区块，否则拒绝
            revert BorrowFreshnessCheck();
        }

        /* Fail gracefully if protocol has insufficient underlying cash */
        if (getCashPrior() < borrowAmount) {
            //如果这个 cToken 池里的底层资产（比如 DAI）不够，就不能借
            revert BorrowCashNotAvailable();
        }

        /*
         * We calculate the new borrower and total borrow balances, failing on overflow:
         *  accountBorrowNew = accountBorrow + borrowAmount
         *  totalBorrowsNew = totalBorrows + borrowAmount
         */
        //更新借款人和全局的借款总额
        uint256 accountBorrowsPrev = borrowBalanceStoredInternal(borrower);
        uint256 accountBorrowsNew = accountBorrowsPrev + borrowAmount;
        uint256 totalBorrowsNew = totalBorrows + borrowAmount;

        /////////////////////////
        // EFFECTS & INTERACTIONS
        // (No safe failures beyond this point)

        /*
         * We write the previously calculated values into storage.
         *  Note: Avoid token reentrancy attacks by writing increased borrow before external transfer.
        `*/
        //先算出新的余额，接着：
        accountBorrows[borrower].principal = accountBorrowsNew;
        accountBorrows[borrower].interestIndex = borrowIndex;
        totalBorrows = totalBorrowsNew;
        //这里更新了：
        // 借款人账户的本金；
        // 借款人当前的利息指数（借款从哪个指数开始算）；
        // 全局总借款量。

        /*
         * We invoke doTransferOut for the borrower and the borrowAmount.
         *  Note: The cToken must handle variations between ERC-20 and ETH underlying.
         *  On success, the cToken borrowAmount less of cash.
         *  doTransferOut reverts if anything goes wrong, since we can't be sure if side effects occurred.
         */
        //将底层资产（比如 DAI）从 cToken 合约中转给借款人
        //这是真正的转钱了!!!!!
        doTransferOut(borrower, borrowAmount);

        /* We emit a Borrow event */
        emit Borrow(borrower, borrowAmount, accountBorrowsNew, totalBorrowsNew); //触发事件,方便前端或区块浏览器监听
    }
```

## Comptroller.`borrowAllowed()`

```solidity
function borrowAllowed(address cToken, address borrower, uint256 borrowAmount)
        external
        override
        returns (uint256)
    {
        // Pausing is a very serious situation - we revert to sound the alarms
        require(!borrowGuardianPaused[cToken], "borrow is paused"); //检查该市场的借款功能是否被 Guardian（紧急开关）暂停

        if (!markets[cToken].isListed) {
            //市场是否被列入
            return uint256(Error.MARKET_NOT_LISTED);
        }

        if (!markets[cToken].accountMembership[borrower]) {
            //账户是否属于该市场（accountMembership），以及自动加入逻辑
            // only cTokens may call borrowAllowed if borrower not in market
            //要求当前调用者 msg.sender 必须正好是该 cToken 合约（原因：只有 cToken 在执行借款流程时才会调用 borrowAllowed；
            //外部不能直接调用 Comptroller.borrowAllowed 来把某人加入市场）
            require(msg.sender == cToken, "sender must be cToken");

            // attempt to add borrower to the market
            //这个步骤通常会把借款人的 assetsIn（账号进入的市场）记录上去，意味着今后该市场的 cToken 可以用作抵押。
            Error err = addToMarketInternal(CToken(msg.sender), borrower);
            if (err != Error.NO_ERROR) {
                return uint256(err);
            }

            // it should be impossible to break the important invariant
            assert(markets[cToken].accountMembership[borrower]);
        }
        //通过 Price Oracle 查询该市场 underlying 的价格（通常是美元价或与基准相关的价格
        if (oracle.getUnderlyingPrice(CToken(cToken)) == 0) {
            //如果价格为 0 → 视为价格异常（可能 oracle 未配置或数据失效），因此拒绝借款并返回 PRICE_ERROR
            return uint256(Error.PRICE_ERROR);
            //如果没有可信价格，协议无法正确计算抵押价值,继续允许借贷会引入严重风险
        }

        uint256 borrowCap = borrowCaps[cToken];
        // borrowCaps 是每个市场可配置的最大借出量（由管理员设置)
        //如果 borrowCap == 0，表示无限制借出
        if (borrowCap != 0) {
            uint256 totalBorrows = CToken(cToken).totalBorrows();
            uint256 nextTotalBorrows = add_(totalBorrows, borrowAmount);
            require(nextTotalBorrows < borrowCap, "market borrow cap reached");
            //否则，取出当前 totalBorrows，加上本次 borrowAmount，若超过 borrowCap 就revert
        }

        //假设性流动性计算（核心：能否借）
        (Error err,, uint256 shortfall) =
            getHypotheticalAccountLiquidityInternal(borrower, CToken(cToken), 0, borrowAmount);
        //模拟 如果 borrower 继续借出 borrowAmount（或同时发生某些 redeem），那么 账户的流动性（liquidity）/短缺（shortfall） 会如何变化
        if (err != Error.NO_ERROR) {
            //函数本身计算出错 (例如计算过程中引用了错误的市场数据等)
            return uint256(err);
        }
        if (shortfall > 0) {
            //说明已经没有足够抵押物支持借款，拒绝
            return uint256(Error.INSUFFICIENT_LIQUIDITY);
        }

        // Keep the flywheel moving
        //即便借款允许，Comptroller 还要在同一时间 更新 COMP 发放的状态
        Exp memory borrowIndex = Exp({mantissa: CToken(cToken).borrowIndex()});
        updateCompBorrowIndex(cToken, borrowIndex); //更新市场的借款奖励指数（累积分发系数）
        distributeBorrowerComp(cToken, borrower, borrowIndex); //把 borrower 当前应得的 COMP 发放或记录下来。
        //好处是：每个用户在任意一次交互（mint/redeem/borrow/repay/…）都会把奖励结算到位，不会出现奖励遗漏或滞后
        return uint256(Error.NO_ERROR);
    }

```

## CToken.`repayBorrowFresh()`

```solidity
 function repayBorrowFresh(address payer, address borrower, uint256 repayAmount) internal returns (uint256) {
        /* Fail if repayBorrow not allowed */
        //Comptroller 权限检查
        uint256 allowed = comptroller.repayBorrowAllowed(address(this), payer, borrower, repayAmount);
        if (allowed != 0) {
            revert RepayBorrowComptrollerRejection(allowed);
        }

        /* Verify market's block number equals current block number */
        if (accrualBlockNumber != getBlockNumber()) {
            //确认区块新鲜度
            revert RepayBorrowFreshnessCheck();
        }

        /* We fetch the amount the borrower owes, with accumulated interest */
        //读取当前借款余额
        uint256 accountBorrowsPrev = borrowBalanceStoredInternal(borrower);

        /* If repayAmount == -1, repayAmount = accountBorrows */
        //确定还款余额

        uint256 repayAmountFinal = repayAmount == type(uint256).max ? accountBorrowsPrev : repayAmount;
        //如果传入 repayAmount = -1（即 uint256.max），代表全额还款
        //否则只还部分
        /////////////////////////
        // EFFECTS & INTERACTIONS
        // (No safe failures beyond this point)

        /*
         * We call doTransferIn for the payer and the repayAmount
         *  Note: The cToken must handle variations between ERC-20 and ETH underlying.
         *  On success, the cToken holds an additional repayAmount of cash.
         *  doTransferIn reverts if anything goes wrong, since we can't be sure if side effects occurred.
         *   it returns the amount actually transferred, in case of a fee.
         */
        //实际转入资产!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
        uint256 actualRepayAmount = doTransferIn(payer, repayAmountFinal);
        //从 payer 转入底层资产（ETH 或 ERC20）
        // 若存在手续费差异（例如某些 ERC20 转账 fee），doTransferIn 会返回实际到账金额

        /*
         * We calculate the new borrower and total borrow balances, failing on underflow:
         *  accountBorrowsNew = accountBorrows - actualRepayAmount
         *  totalBorrowsNew = totalBorrows - actualRepayAmount
         */
        //更新借款账本
        uint256 accountBorrowsNew = accountBorrowsPrev - actualRepayAmount;
        uint256 totalBorrowsNew = totalBorrows - actualRepayAmount;

        /* We write the previously calculated values into storage */
        accountBorrows[borrower].principal = accountBorrowsNew;
        accountBorrows[borrower].interestIndex = borrowIndex;
        totalBorrows = totalBorrowsNew;
        //借款人余额减少
        // 协议全局总借款减少
        // 保存最新借款指数 borrowIndex 以便下次利息计算

        /* We emit a RepayBorrow event */
        emit RepayBorrow(payer, borrower, actualRepayAmount, accountBorrowsNew, totalBorrowsNew);

        return actualRepayAmount;
    }
```

# 五、清算 liquidation

清算人（liquidator）在借款人\*\*触发 shortfall（保证金不足）\*\*时，有权：

1. **偿还部分债务（repay）**
2. **按一定折价（liquidation incentive）获得抵押品（seize）**

```solidity
        nonReentrant
```

## ctoken.`liquidateBorrowFresh()`

```solidity
function liquidateBorrowFresh(
        address liquidator,
        address borrower,
        uint256 repayAmount,
        CTokenInterface cTokenCollateral
    ) internal {
        /* Fail if liquidate not allowed */
        uint256 allowed = comptroller.liquidateBorrowAllowed(
            address(this), address(cTokenCollateral), liquidator, borrower, repayAmount
        );
        if (allowed != 0) {
            revert LiquidateComptrollerRejection(allowed);
        }

        /* Verify market's block number equals current block number */
        //要求两个市场都刚刚调用过 accrueInterest()
        if (accrualBlockNumber != getBlockNumber()) {
            revert LiquidateFreshnessCheck();
        }

        /* Verify cTokenCollateral market's block number equals current block number */
        ////要求两个市场都刚刚调用过 accrueInterest()
        if (cTokenCollateral.accrualBlockNumber() != getBlockNumber()) {
            revert LiquidateCollateralFreshnessCheck();
        }

        /* Fail if borrower = liquidator */
        //清算人不能自己清算自己
        if (borrower == liquidator) {
            revert LiquidateLiquidatorIsBorrower();
        }

        /* Fail if repayAmount = 0 */
        //不能传0金额
        if (repayAmount == 0) {
            revert LiquidateCloseAmountIsZero();
        }

        /* Fail if repayAmount = -1 */
        //不能传 uint256.max（那是全额还款的特殊标识）
        if (repayAmount == type(uint256).max) {
            revert LiquidateCloseAmountIsUintMax();
        }

        /* Fail if repayBorrow fails */
        //清算人转钱的一步,帮欠钱的减少债务
        uint256 actualRepayAmount = repayBorrowFresh(liquidator, borrower, repayAmount);

        /////////////////////////
        // EFFECTS & INTERACTIONS
        // (No safe failures beyond this point)

        /* We calculate the number of collateral tokens that will be seized */
        (uint256 amountSeizeError, uint256 seizeTokens) =
            comptroller.liquidateCalculateSeizeTokens(address(this), address(cTokenCollateral), actualRepayAmount);
        require(amountSeizeError == NO_ERROR, "LIQUIDATE_COMPTROLLER_CALCULATE_AMOUNT_SEIZE_FAILED");

        /* Revert if borrower collateral token balance < seizeTokens */
        require(cTokenCollateral.balanceOf(borrower) >= seizeTokens, "LIQUIDATE_SEIZE_TOO_MUCH");

        // If this is also the collateral, run seizeInternal to avoid re-entrancy, otherwise make an external call
        if (address(cTokenCollateral) == address(this)) {
            //如果借款和抵押在同一个市场（例如借的 cETH 清 cETH），就直接调用 seizeInternal()
            seizeInternal(address(this), liquidator, borrower, seizeTokens);
        } else {
            //如果借款和抵押是不同市场，就调用另一市场的 seize() 函数
            require(cTokenCollateral.seize(liquidator, borrower, seizeTokens) == NO_ERROR, "token seizure failed");
        }

        /* We emit a LiquidateBorrow event */
        emit LiquidateBorrow(liquidator, borrower, actualRepayAmount, address(cTokenCollateral), seizeTokens);
    }

```

## comptroller.`liquidateBorrowAllowed()`

```solidity
function liquidateBorrowAllowed(
        address cTokenBorrowed,
        address cTokenCollateral,
        address liquidator,
        address borrower,
        uint256 repayAmount
    ) external override returns (uint256) {
        // Shh - currently unused
        liquidator;

        if (!markets[cTokenBorrowed].isListed || !markets[cTokenCollateral].isListed) {
            return uint256(Error.MARKET_NOT_LISTED); //两个市场是否都上市
        }

        uint256 borrowBalance = CToken(cTokenBorrowed).borrowBalanceStored(borrower);

        /* allow accounts to be liquidated if the market is deprecated */
        if (isDeprecated(CToken(cTokenBorrowed))) {
            require(borrowBalance >= repayAmount, "Can not repay more than the total borrow");
        } else {
            /* The borrower must have shortfall in order to be liquidatable */
            (Error err,, uint256 shortfall) = getAccountLiquidityInternal(borrower);
            if (err != Error.NO_ERROR) {
                return uint256(err);
            }

            if (shortfall == 0) {
                //借款人确实有 shortfall（抵押不足）
                return uint256(Error.INSUFFICIENT_SHORTFALL);
            }

            /* The liquidator may not repay more than what is allowed by the closeFactor */
            uint256 maxClose = mul_ScalarTruncate(Exp({mantissa: closeFactorMantissa}), borrowBalance);
            if (repayAmount > maxClose) {
                //清算人不能清算超过 closeFactor 限制
                return uint256(Error.TOO_MUCH_REPAY);
            }
        }
        return uint256(Error.NO_ERROR);
    }
```

## ctoken.`repayBorrowFresh()`

```solidity
function repayBorrowFresh(address payer, address borrower, uint256 repayAmount) internal returns (uint256) {
        /* Fail if repayBorrow not allowed */
        //Comptroller 权限检查
        uint256 allowed = comptroller.repayBorrowAllowed(address(this), payer, borrower, repayAmount);
        if (allowed != 0) {
            revert RepayBorrowComptrollerRejection(allowed);
        }

        /* Verify market's block number equals current block number */
        if (accrualBlockNumber != getBlockNumber()) {
            //确认区块新鲜度
            revert RepayBorrowFreshnessCheck();
        }

        /* We fetch the amount the borrower owes, with accumulated interest */
        //读取当前借款余额
        uint256 accountBorrowsPrev = borrowBalanceStoredInternal(borrower);

        /* If repayAmount == -1, repayAmount = accountBorrows */
        //确定还款余额

        uint256 repayAmountFinal = repayAmount == type(uint256).max ? accountBorrowsPrev : repayAmount;
        //如果传入 repayAmount = -1（即 uint256.max），代表全额还款
        //否则只还部分
        /////////////////////////
        // EFFECTS & INTERACTIONS
        // (No safe failures beyond this point)

        /*
         * We call doTransferIn for the payer and the repayAmount
         *  Note: The cToken must handle variations between ERC-20 and ETH underlying.
         *  On success, the cToken holds an additional repayAmount of cash.
         *  doTransferIn reverts if anything goes wrong, since we can't be sure if side effects occurred.
         *   it returns the amount actually transferred, in case of a fee.
         */
        //实际转入资产!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
        uint256 actualRepayAmount = doTransferIn(payer, repayAmountFinal);
        //从 payer 转入底层资产（ETH 或 ERC20）
        // 若存在手续费差异（例如某些 ERC20 转账 fee），doTransferIn 会返回实际到账金额

        /*
         * We calculate the new borrower and total borrow balances, failing on underflow:
         *  accountBorrowsNew = accountBorrows - actualRepayAmount
         *  totalBorrowsNew = totalBorrows - actualRepayAmount
         */
        //更新借款账本
        uint256 accountBorrowsNew = accountBorrowsPrev - actualRepayAmount;
        uint256 totalBorrowsNew = totalBorrows - actualRepayAmount;

        /* We write the previously calculated values into storage */
        accountBorrows[borrower].principal = accountBorrowsNew;
        accountBorrows[borrower].interestIndex = borrowIndex;
        totalBorrows = totalBorrowsNew;
        //借款人余额减少
        // 协议全局总借款减少
        // 保存最新借款指数 borrowIndex 以便下次利息计算

        /* We emit a RepayBorrow event */
        emit RepayBorrow(payer, borrower, actualRepayAmount, accountBorrowsNew, totalBorrowsNew);

        return actualRepayAmount;
    }
```

## comptroller. `liquidateCalculateSeizeTokens()`

```solidity
seizeTokens = actualRepayAmount × (liquidationIncentive × priceBorrowed) / (priceCollateral × exchangeRate)
```

* `liquidationIncentive`: 清算奖励（通常是 1.08，即多拿 8%）
* `priceBorrowed`: 借款资产价格
* `priceCollateral`: 抵押品价格
* `exchangeRate`: 抵押品市场的兑换率（1 cToken = ? underlying）

```solidity
function liquidateCalculateSeizeTokens(address cTokenBorrowed, address cTokenCollateral, uint256 actualRepayAmount)
        external
        view
        override
        returns (uint256, uint256)
    {
        /* Read oracle prices for borrowed and collateral markets */
        uint256 priceBorrowedMantissa = oracle.getUnderlyingPrice(CToken(cTokenBorrowed));
        uint256 priceCollateralMantissa = oracle.getUnderlyingPrice(CToken(cTokenCollateral));
        if (priceBorrowedMantissa == 0 || priceCollateralMantissa == 0) {
            return (uint256(Error.PRICE_ERROR), 0);
        }

        /*
         * Get the exchange rate and calculate the number of collateral tokens to seize:
         *  seizeAmount = actualRepayAmount * liquidationIncentive * priceBorrowed / priceCollateral
         *  seizeTokens = seizeAmount / exchangeRate
         *   = actualRepayAmount * (liquidationIncentive * priceBorrowed) / (priceCollateral * exchangeRate)
         */
        /*

        *获取汇率并计算要扣押的抵押代币数量：

        *扣押金额=实际还款金额*清算激励*价格借款/价格抵押品

        *扣押代币=扣押金额/汇率

        * = 实际还款金额*（清偿激励*借款价格）/（价格抵押品*汇率）

        */
        uint256 exchangeRateMantissa = CToken(cTokenCollateral).exchangeRateStored(); // Note: reverts on error
        uint256 seizeTokens;
        Exp memory numerator;
        Exp memory denominator;
        Exp memory ratio;

        numerator = mul_(Exp({mantissa: liquidationIncentiveMantissa}), Exp({mantissa: priceBorrowedMantissa}));
        denominator = mul_(Exp({mantissa: priceCollateralMantissa}), Exp({mantissa: exchangeRateMantissa}));
        ratio = div_(numerator, denominator);

        seizeTokens = mul_ScalarTruncate(ratio, actualRepayAmount);

        return (uint256(Error.NO_ERROR), seizeTokens);
    }
```

```solidity
function seizeInternal(address seizerToken, address liquidator, address borrower, uint256 seizeTokens) internal {
  /* Fail if seize not allowed */
  uint256 allowed = comptroller.seizeAllowed(address(this), seizerToken, liquidator, borrower, seizeTokens);
  if (allowed != 0) {
    //allowed例行检查
    revert LiquidateSeizeComptrollerRejection(allowed);
  }

  /* Fail if borrower = liquidator */
  //禁止自己清算自己
  if (borrower == liquidator) {
    revert LiquidateSeizeLiquidatorIsBorrower();
  }

  /*
         * We calculate the new borrower and liquidator token balances, failing on underflow/overflow:
         *  borrowerTokensNew = accountTokens[borrower] - seizeTokens
         *  liquidatorTokensNew = accountTokens[liquidator] + seizeTokens
         */
  uint256 protocolSeizeTokens = mul_(seizeTokens, Exp({mantissa: protocolSeizeShareMantissa})); //给协议的部分奖励
  uint256 liquidatorSeizeTokens = seizeTokens - protocolSeizeTokens; //给清算人的部分奖励
  Exp memory exchangeRate = Exp({mantissa: exchangeRateStoredInternal()}); //把协议的这部分换算成标的资产价值,exchangerate是每个Ctoken对应的实际资产价值
  uint256 protocolSeizeAmount = mul_ScalarTruncate(exchangeRate, protocolSeizeTokens); //清算人 cToken 增加
  uint256 totalReservesNew = totalReserves + protocolSeizeAmount; //协议储备 totalReserves 增加

  /////////////////////////
  // EFFECTS & INTERACTIONS
  // (No safe failures beyond this point)

  /* We write the calculated values into storage */
  totalReserves = totalReservesNew;
  totalSupply = totalSupply - protocolSeizeTokens;
  accountTokens[borrower] = accountTokens[borrower] - seizeTokens; //借款人抵押品减少
  accountTokens[liquidator] = accountTokens[liquidator] + liquidatorSeizeTokens; //清算人 cToken 增加

  /* Emit a Transfer event */
  emit Transfer(borrower, liquidator, liquidatorSeizeTokens);
  emit Transfer(borrower, address(this), protocolSeizeTokens);
  emit ReservesAdded(address(this), protocolSeizeAmount, totalReservesNew);
}
```
