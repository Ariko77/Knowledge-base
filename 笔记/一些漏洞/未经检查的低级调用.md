# 未经检查的低级调用

# 低级调用

<font style="color:rgb(44, 62, 80);">以太坊的低级调用包括 </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">call()</font></code><font style="color:rgb(44, 62, 80);">，</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">delegatecall()</font></code><font style="color:rgb(44, 62, 80);">，</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">staticcall()</font></code><font style="color:rgb(44, 62, 80);">，和</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">send()</font></code>

# <font style="color:rgb(44, 62, 80);">send</font>

<font style="color:rgb(44, 62, 80);">一些合约使用 </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">send()</font></code><font style="color:rgb(44, 62, 80);"> 发送 </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">ETH</font></code><font style="color:rgb(44, 62, 80);">，但是 </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">send()</font></code><font style="color:rgb(44, 62, 80);"> 限制 gas 要低于 2300，否则会失败。当目标地址的回调函数比较复杂时，花费的 gas 将高于 2300，从而导致 </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">send()</font></code><font style="color:rgb(44, 62, 80);"> 失败。如果此时在上层函数没有检查返回值的话，交易继续执行，就会出现意想不到的问题。</font>

## <font style="color:rgb(44, 62, 80);">漏洞例子</font>

### [<font style="color:rgb(62, 175, 124);">#</font>](http://x2d.co/solidity/safe/UncheckedCall.html#%E9%93%B6%E8%A1%8C%E5%90%88%E7%BA%A6)<font style="color:rgb(44, 62, 80);">银行合约</font>

<font style="color:rgb(44, 62, 80);">这个合约是在</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">S01 重入攻击</font></code><font style="color:rgb(44, 62, 80);">教程中的银行合约基础上修改而成。它包含</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">1</font></code><font style="color:rgb(44, 62, 80);">个状态变量</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">balanceOf</font></code><font style="color:rgb(44, 62, 80);">记录所有用户的以太坊余额；并且包含</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">3</font></code><font style="color:rgb(44, 62, 80);">个函数：</font>

* <code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">deposit()</font></code><font style="color:rgb(44, 62, 80);">：存款函数，将</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">ETH</font></code><font style="color:rgb(44, 62, 80);">存入银行合约，并更新用户的余额。</font>
* <code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">withdraw()</font></code><font style="color:rgb(44, 62, 80);">：提款函数，将调用者的余额转给它。具体步骤和上面故事中一样：查询余额，更新余额，转账。</font>**<font style="color:rgb(44, 62, 80);">注意：这个函数没有检查</font>\*\*\*\*<font style="color:rgb(44, 62, 80);"> </font>**<code>**<font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">send()</font>**</code>**<font style="color:rgb(44, 62, 80);"> </font>\*\*\*\*<font style="color:rgb(44, 62, 80);">的返回值，提款失败但余额会清零！</font>**
* <code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">getBalance()</font></code><font style="color:rgb(44, 62, 80);">：获取银行合约里的</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">ETH</font></code><font style="color:rgb(44, 62, 80);">余额。</font>

```solidity
contract UncheckedBank {
    mapping (address => uint256) public balanceOf;    // 余额mapping

    // 存入ether，并更新余额
    function deposit() external payable {
        balanceOf[msg.sender] += msg.value;
    }

    // 提取msg.sender的全部ether
    function withdraw() external {
        // 获取余额
        uint256 balance = balanceOf[msg.sender];
        require(balance > 0, "Insufficient balance");
        balanceOf[msg.sender] = 0;
        // Unchecked low-level call
        bool success = payable(msg.sender).send(balance);
    }

    // 获取银行合约的余额
    function getBalance() external view returns (uint256) {
        return address(this).balance;
    }
}
```

## <font style="color:rgb(44, 62, 80);">攻击合约</font>

<font style="color:rgb(44, 62, 80);">我们构造了一个攻击合约，它刻画了一个倒霉的储户，取款失败但是银行余额清零：合约回调函数</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">receive()</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">中的</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">revert()</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">将回滚交易，因此它无法接收</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">ETH</font></code><font style="color:rgb(44, 62, 80);">；但是提款函数</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">withdraw()</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">却能正常调用，清空余额。</font>

```solidity
contract Attack {
    UncheckedBank public bank; // Bank合约地址

    // 初始化Bank合约地址
    constructor(UncheckedBank _bank) {
        bank = _bank;
    }
    
    // 回调函数，转账ETH时会失败
    receive() external payable {
        revert();
    }

    // 存款函数，调用时 msg.value 设为存款数量
    function deposit() external payable {
        bank.deposit{value: msg.value}();
    }

    // 取款函数，虽然调用成功，但实际上取款失败
    function withdraw() external payable {
        bank.withdraw();
    }

    // 获取本合约的余额
    function getBalance() external view returns (uint256) {
        return address(this).balance;
    }
}
```

## <code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">Remix</font></code><font style="color:rgb(44, 62, 80);">复现</font>

1. <font style="color:rgb(44, 62, 80);">部署</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">UncheckedBank</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">合约。</font>
2. <font style="color:rgb(44, 62, 80);">部署</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">Attack</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">合约，构造函数填入</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">UncheckedBank</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">合约地址。</font>
3. <font style="color:rgb(44, 62, 80);">调用</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">Attack</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">合约的</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">deposit()</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">存款函数，存入</font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">1 ETH</font></code><font style="color:rgb(44, 62, 80);">。</font>
4. <font style="color:rgb(44, 62, 80);">调用</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">Attack</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">合约的</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">withdraw()</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">提款函数，进行提款，调用成功。</font>
5. <font style="color:rgb(44, 62, 80);">分别调用</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">UncheckedBank</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">合约的</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">balanceOf()</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">函数和</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">Attack</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">合约的</font><font style="color:rgb(44, 62, 80);"> </font><code><font style="color:rgb(71, 101, 130);background-color:rgba(27, 31, 35, 0.05);">getBalance()</font></code><font style="color:rgb(44, 62, 80);"> </font><font style="color:rgb(44, 62, 80);">函数。尽管上一步调用成功并且储户余额被清空，但是提款失败了。</font>
