# 源码
```solidity
// SPDX-License-Identifier: SEE LICENSE IN LICENSE
pragma solidity ^0.8.0;

//0x74aF5fc21e39600BeB29f59004a0861aB94deBA1

contract MagicBox {
    struct Signature {
        uint8 v;
        bytes32 r;
        bytes32 s;
    }

    address magician;
    bytes32 alreadyUsedSignatureHash;
    bool isOpened;

    constructor() {}

    function isSolved() public view returns (bool) {
        return isOpened;
    }

    //生成一个消息哈希

    function getMessageHash(address _magician) public view returns (bytes32) {
        return
            keccak256(
                abi.encodePacked(
                    "I want to open the magic box",
                    _magician,
                    address(this),
                    block.chainid
                )
            );
    } //978262f95ee4e354f94cdcc95c9219b488b7dc67d404b9f122803e8e6d1130a4

    // 签名 ：ed78dbee09530c31b12179aab0e65c71cb84762c57455dbcf5af92ef70d42e10 6b8ff96129a228534c326eed8eeada97817a95cf8116dbda07df271636a93049 1c

    function _getSignerAndSignatureHash(
        Signature memory _signature
    ) internal view returns (address, bytes32) {
        address signer = ecrecover(
            getMessageHash(msg.sender),
            _signature.v,
            _signature.r,
            _signature.s
        );
        bytes32 signatureHash = keccak256(
            abi.encodePacked(_signature.v, _signature.r, _signature.s)
        );
        return (signer, signatureHash);
    }

    function signIn(Signature memory signature) external {
        require(magician == address(0), "Magician already signed in"); //只能有一个人报名
        (address signer, bytes32 signatureHash) = _getSignerAndSignatureHash(
            signature
        ); // 调用者传入签名的签名地址和哈希
        require(signer == msg.sender, "Invalid signature"); //签名者必须是调用者
        magician = signer; //调用者成为魔术师
        alreadyUsedSignatureHash = signatureHash; //已经使用的签名哈希，方式重放攻击
    }

    function openBox(Signature memory signature) external {
        require(magician == msg.sender, "Only magician can open the box"); //need sign in，此时alreadyUsedSignatureHash==msg.sender
        (address signer, bytes32 signatureHash) = _getSignerAndSignatureHash(
            signature
        ); //调用_getSignerAndSignatureHash(),得到msg.sender的signer signatureHash
        require(signer == msg.sender, "Invalid signature"); //ok
        require(
            signatureHash != alreadyUsedSignatureHash,
            "Signature already used"
        );
        isOpened = true;
    }
}

```

# 思路
这脚本用到了作弊码 ，就这么看看吧

1. 调用getMessageHash（eoa地址），得到一个消息哈希
2. 用得到的v，r，s签名来signin，就相当于给eoa报名成为魔术师了,此时：
+ signatureHash = keccak256(v1, r1, s1) 
+  alreadyUsedSignatureHash = signatureHash  
3. 这个openBoxHash和第一步那个signInHash其实是一个，就是方便区分
4. 然后找个newS，也就是N-s
5. 用v，r，newS去openbox，这个时候也是可以打开，因为两组vrs都指向同一个eoa地址

此时

+  signatureHash = keccak256(v2, r2, newS) 

满足了这个require条件

```solidity
            signatureHash != alreadyUsedSignatureHash,
```
